GITHUB_PRS_RESPONSE=$(curl -s \
    -H "Accept: application/vnd.github+json" \
    -H "Authorization: Bearer $GITHUB_TOKEN" \
    -H "X-GitHub-Api-Version: 2022-11-28" \
    https://api.github.com/repos/Kotlin/kotlinx.serialization/pulls\?state\=all\&per_page\=100\&page\=1)

echo "$GITHUB_PRS_RESPONSE" | jq -r '["id","created_at","closed_at"], (.[] | [.id, .created_at, .closed_at]) | @csv'
